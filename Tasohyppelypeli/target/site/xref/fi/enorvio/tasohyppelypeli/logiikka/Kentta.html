87      public int selvitaPikselinTyyppi(int x, int y) {
88          int sarake = x / 16;
89          int rivi = y / 16;
90          return this.laatat[rivi][sarake];
91      }
92  
93      public void poistaPiste(int x, int y) {
94          if (selvitaPikselinTyyppi(x, y) == piste) {
95              int sarake = x / 16;
96              int rivi = y / 16;
97              if (Math.abs(x - (16 * sarake + 6)) < 10) {
98                  if ((Math.abs(y - (16 * rivi + 6)) < 10)) {
99                      this.setLaatta(sarake, rivi, tyhja);
100                     this.pisteet++;
101                 }
102             }
103         }
104 
105     }

107     public void luoTeleportti(int alkux, int alkuy, int loppux, int loppuy) {
108         int[] alku = {alkux, alkuy};
109         int[] loppu = {loppux, loppuy};
110         this.teleportit.put(alku, loppu);
111         this.teleportit.put(loppu, alku);
112         this.setLaatta(alkux, alkuy, 4);
113         this.setLaatta(loppux, loppuy, 4);
114     }
115 
116     public void tulostaTeleportit() {
117         for (int[] alku : this.teleportit.keySet()) {
118             System.out.println("alku: " + alku[0] + ", " + alku[1] + ", loppu: " + this.teleportit.get(alku)[0] + ", " + this.teleportit.get(alku)[1]);
119         }
120     }
121 
122     public int[] getTeleportinToinenPaa(int[] alku) {
123         for (int[] taulukko : this.teleportit.keySet()) {
124             if (Arrays.equals(taulukko, alku)) {
125                 return this.teleportit.get(taulukko);
126             }
127         }
128         return null;
129     }
130 
131     public HashMap<int[], int[]> getTeleportit() {
132         return this.teleportit;
133     }
