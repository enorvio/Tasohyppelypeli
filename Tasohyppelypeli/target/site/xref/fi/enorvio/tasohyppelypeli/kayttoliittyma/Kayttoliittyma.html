25  /**
26   * Luokka sisältää pelin aloitusvalikon.
27   */
28  public class Kayttoliittyma implements Runnable {
29  
30      private JFrame ikkuna;
31      private JLabel statuspaneeli;
32      private GridLayout asettelija;
33      private JButton uusiPeli;
34      private JButton lopeta;
35      private JButton korkeimmatPisteet;
36  
37      public Kayttoliittyma() {
38  
39      }
40  
41      @Override
42      public void run() {
43          ikkuna = new JFrame("Valikko");
44          ikkuna.setPreferredSize(new Dimension(600, 400));
45          ikkuna.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
46  
47          luoKomponentit(ikkuna.getContentPane());
48  
49          ikkuna.pack();
50          ikkuna.setVisible(true);
51      }
52  
53      /**
54       * Metodi luo aloitusvalikon komponentit, asettaa ne aloitusvalikkoon, luo niille kuuntelijan ja kytkee sen komponentteihin.
55       *
56       * @param container Container-olio, johon komponentit sijoitetaan
57       */ 
58      private void luoKomponentit(Container container) {
59          GridLayout asettelija = new GridLayout(3, 1);
60          container.setLayout(asettelija);
61          JPanel paneeli = new JPanel(new GridLayout(1, 3));
62          paneeli.setFocusable(true);
63          JButton nappi1 = new JButton("New game");
64          paneeli.add(nappi1);
65          JButton nappi3 = new JButton("High score");
66          paneeli.add(nappi3);
67          JButton nappi4 = new JButton("Quit");
68          paneeli.add(nappi4);
69          container.add(paneeli);
70          ValikonKuuntelija kuuntelija = (new ValikonKuuntelija(ikkuna, nappi1, nappi3, nappi4));
71          nappi1.addActionListener(kuuntelija);
72          nappi3.addActionListener(kuuntelija);
73          nappi4.addActionListener(kuuntelija);
74  
75      }
