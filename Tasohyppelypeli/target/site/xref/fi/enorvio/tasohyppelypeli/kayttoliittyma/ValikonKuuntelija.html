19  /**
20   * Luokassa on toteutettu aloitusvalikon nappien painamiseen reagointi, mukaan lukien uuden pelin aloittaminen.
21   */
22  public class ValikonKuuntelija implements ActionListener {
23  
24      private JButton uusiPeli;
25      private JButton lataaPeli;
26      private JButton lopeta;
27      private JButton korkeimmatPisteet;
28      
29      
30      public ValikonKuuntelija(JButton uusiPeli, JButton lataaPeli, JButton lopeta, JButton korkeimmatPisteet) {
31          this.uusiPeli = uusiPeli;
32          this.lataaPeli = lataaPeli;
33          this.lopeta = lopeta;
34          this.korkeimmatPisteet = korkeimmatPisteet;
35      }
36  
37      @Override
38      public void actionPerformed(ActionEvent ae) {
39  
40          try {
41              if (ae.getSource() == uusiPeli) {
42                  System.out.println("jee");
43                  String[] kentanNimet = {"src/main/resources/esimerkkiTaso2.txt", "src/main/resources/esimerkkiTaso.txt"};
44                  Logiikka logiikka = new Logiikka(kentanNimet);
45                  PeliIkkuna esimerkkiPeliIkkuna = new PeliIkkuna(logiikka);
46                  System.out.println("uusi peli");
47                  JFrame ikkuna = new JFrame("Peli");
48                  GridLayout asettelija = new GridLayout(2, 1);
49                  ikkuna.setLayout(asettelija);
50                  ikkuna.add(esimerkkiPeliIkkuna);
51                  JLabel kentta = esimerkkiPeliIkkuna.getStatuspaneeli();
52                  ikkuna.add(kentta);
53                  ikkuna.setPreferredSize(new Dimension(600, 600));
54                  ikkuna.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
55                  ikkuna.pack();
56                  ikkuna.setVisible(true);
57              }
58              if (ae.getSource() == lopeta) {
59                  System.out.println("jee");
60              }
61          } catch (Exception e) {
62              System.out.println(e);
63              System.out.println("virhe");
64          }
65  
66      }

68  }
